# PDF Processing
# ------------------------------------------------------------------------------
# PyMuPDF (fitz) is NOT used because its binary wheel (cp310-abi3-win_amd64)
# links against internal CPython symbols that were removed in CPython 3.14,
# causing an ImportError at startup.  pypdf is a pure-Python replacement
# (wheel tag: py3-none-any) that works on every CPython version ≥ 3.8.
pypdf>=5.0.0                # Pure-Python PDF parser — extracts text, metadata,
                            # and page count from uploaded research papers.
                            # No binary extensions; compatible with Python 3.14+.
